/**
 * @param {number[]} nums
 * @param {number} val
 * @return {number}
 */
var removeElement = function(nums, val) {
    var numsResult =[];

    for (i=0; i<nums.length;i++){

        if (nums[i]!=val){
            numsResult.push(nums[i]);
        }
    }
    console.log(numsResult);

    nums.splice(0,nums.length);
    nums.push.apply(nums,numsResult);
};
