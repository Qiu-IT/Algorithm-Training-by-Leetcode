/**
 * @param {number} x
 * @return {boolean}
 */
var isPalindrome = function(x) {
 var strOriginal = x.toString();
 var strReverse = "";
 for (let i=strOriginal.length-1; i>=0;i-- ) {

        strReverse += strOriginal[i];
 }

    if (strOriginal == strReverse ) {

        return true;
    }

    else {

        return false;
    }
 
};
